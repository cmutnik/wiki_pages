import scipy.optimize as optimized # needed to fit

def linfit(x, m, b):
	return(m*x + b)

def gaussian(x, a, b, c, d):
    """
    Gaussian fit
    a = peak height, b = peak center, c = width of peak, d = continuum
    """
    y = a * np.exp((-(x-b)**2)/(2*c**2)) + d
    return y

import scipy.optimize as optimized # needed to fit
import matplotlib.pyplot as plt
from astropy.io import ascii
import numpy as np

def linfit(x, m, b):
	return(m*x + b)

#dfile=ascii.read('period_allmask_limit15.txt.chyea')
#gmag = dfile['col6']
#s = np.array(gmag)

per15stars = ascii.read('starlists/period_allmask_limit15.txt')
nonper15stars = ascii.read('starlists/nonperiod_allmask_limit15.txt')

per_gmag = per15stars['6.g_Mean_mag']
nonper_gmag = nonper15stars['6.g_Mean_mag']

# make list of all stars mean g mag
allstar_gmag = np.append(per_gmag,nonper_gmag)

s = np.array(allstar_gmag)
xmin = 0
xmax = 20
step = 0.2
y, x = np.histogram(s, bins=np.linspace(xmin, xmax, (xmax-xmin)/step))
nbins = y.size

def plot_histogram():
	plt.clf()
	fig, ax = plt.subplots()
	plt.bar(x[:-1], y, width=x[1]-x[0], color='green', alpha=0.5)
	#plt.hist(s, bins=nbins, alpha=0.5)

	plt.xlabel('$m_g$')
	plt.ylabel('N')
	plt.yscale('log')
	plt.xlim(8.5,18)
	#plt.xlim(9.2,17.2)
	#plt.ylim(0,200)
	fig.subplots_adjust(left=0.125)
	#plt.show()
	plt.savefig('log_with_step_0.2_newlim.png')
plot_histogram()

# final value where x is less than 13.3
