Generated observation script using 

  awk -v mjd=57452.2226 -v ovhd=10 'BEGIN{obsnum=1} \
    NR>1{obj=$1; ra=$2; dec=$3; filt=$4; dt=$5; \
       zt=mjd-int(mjd)+(24-10)/24; zh=int(zt*24); zm=int((zt-zh/24)*24*60);
       printf "%5d%04d %6.2f %6.2f %2s %5.1f %s %5.4f %4d:%02d 0:0   0 %4d\n", int(mjd), obsnum, ra, dec, filt, dt, $1, mjd, zh, zm, ovhd; \
       mjd+=(dt+ovhd)/86400; obsnum=obsnum+1}' observation_list_RR_Lyrae_160304.txt > /tmp/foo.obslist

  cat /tmp/foo.obslist | while read LINE ; do
    m=$(echo $LINE | awk '{print $6}')
    a=$(echo $LINE | awk '{print $2}')
    d=$(echo $LINE | awk '{print $3}')
    eval $(skyangle mjd=$m ra=$a dec=$d)
    echo $ha $alt $skymu | awk '{printf "%6d %4.1f %4.1f  000   00\n",$1,$2,$3}'
  done > /tmp/foo.halt

  echo "#ID         RA    Dec filt exptime Object  MJD-UTC    tzone i:N  tti  ovhd    HA Alt  Sky  Domaz Domtol" > rrlyr_sched.57452
  paste -d ' ' /tmp/foo.obslist /tmp/foo.halt >> rrlyr_sched.57452

Sort for all RR Lyrae that are in overlapping fields 

  mkdir /home/rr_lyrae/tmp/CM ; cd /home/rr_lyrae/tmp/CM
# All non-bad observations
  for m in {57396..57460} ; do
    awk '$5>=20 && $8<3 && $8!=0 && $9>500 && $10>15 && $11!=0 && $12!=0{print $0}' /atlas/red/01p/$m/$m.log
  done > all.log
# All observations that overlap a point to within radius RAD=2.5 + extra
  RA0=100.5 DEC0=-0.5 RAD=3.0
  echo $RA0 $DEC0 | cm 2,1 - 12,11 all.log -tol $RAD,d | awk '{print $4}' > list.$RA0.$DEC0
# Pick detections from a particular 1 deg region
  RA1=99.0 RA2=101.0 DEC1=-1.0 DEC2=0.0
  rm hunk.$RA1.$RA2.$DEC1.$DEC2
  for f in $(cat list.$RA0.$DEC0) ; do
    m=$(echo $f | tr '[a-z]' ' ' | awk '{print $2}')
    obs=$(echo $f | awk '{print substr($1,1,14)}')
    dev=$(echo $f | awk '{print substr($1,15,1)}')
    filt=$(echo $f | awk '{print substr($1,14,1)}')
    awk -v a1=$RA1 -v a2=$RA2 -v d1=$DEC1 -v d2=$DEC2 -v obs=$obs -v dev=$dev -v filt=$filt 'NR>1 && $1>=a1 && $1<=a2 && $2>=d1 && $2<=d2{printf "%s%d %s\n", obs, dev, $0}' /atlas/red/01p/$m/$f.dph >> hunk.$RA1.$RA2.$DEC1.$DEC2
    echo $obs $dev $filt
  done
# Group each filter
  for filt in g r i ; do
    grep $filt hunk.$RA1.$RA2.$DEC1.$DEC2 > hunk.$filt
    cm 3,2,4 hunk.$filt -tol 0.001,d,2 -grp -grpfmt "%8.4f" -grpfile grp.$filt > grpall.$filt
    awk -v nmin=12 '$2>nmin{print $0}' grp.$filt > star.$filt
  done

  • this reduced 1300 deg2 observation data down to ~300

  • sorted biggest (most variable?) to smallest (least variable?)
  • ran LS on 90,000 most variable stars, rather than full star groupings (over 1million)
