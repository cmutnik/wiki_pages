    import numpy as np
import matplotlib.pyplot as plt
from matplotlib.transforms import Bbox
import matplotlib.path as mpath

# load data from galaxy with app mag in V (Vega)
dfiles = np.genfromtxt('M13.txt')
app_mag = dfiles[:,3]
app_mag_err = dfiles[:,4]
gal_VI = dfiles[:,5]
gal_VI_err = dfiles[:,6]

# load data from isochrone
isodat = np.genfromtxt('iso1.txt')
iso_VI = isodat[:,7] - isodat[:,9]
iso_mag_DS1 = isodat[:,-1]

# used to determine distance
dist = 7695.695 #known distance
abs_mag = app_mag - (5.*np.log10(dist/10.))

# turn points into coulmns
gal_points = np.column_stack([gal_VI, abs_mag])

# plot last point of iso (lowest mag)
dy_box = 2.5/2.
dx_box = 0.75/2.

# choose point for box to go around
box_center = gal_points[0]

# Box corners
LTC = [box_center[0]-dx_box , box_center[1]-dy_box ]
RTC = [box_center[0]+dx_box , box_center[1]-dy_box ]
LBC = [box_center[0]-dx_box , box_center[1]+dy_box ]
RBC = [box_center[0]+dx_box , box_center[1]+dy_box ]

# extra comma after RTC[0] is needed or all points wont get enclosed
verts = np.array([[LTC[0], LBC[0], RBC[0], RTC[0], ], [LTC[1], LBC[1], RBC[1], RTC[1]]]).T

# draw path around verts to enclose points
path = mpath.Path(verts)
points_inside = gal_points[path.contains_points(gal_points)]
N_stars_inside_box = len(points_inside)
print 'Number of stars in box: ' , N_stars_inside_box

# Plot data, box, and identify points in box
plt.figure()
plt.gca().invert_yaxis()

#!/usr/bin/python
# Py script to make a histogrm of stars in globular cluster, for removal of outliers
# Corey Mutnik 5/4/16
import numpy as np
import matplotlib.pyplot as plt
from matplotlib import cm
import scipy.optimize as optimized
import sys

# count number of system inputs
total = len(sys.argv)

# input value on command line, with default value incase you forget
if total < 2:
	sigma = 3
else:
	# take input as sigma value: python [filename.py] [sigma]
	sigma=float(sys.argv[1])

# import data
dfile = np.genfromtxt('M3_lcogt_VI_no_dubs.txt')

# not sure if these columns are correct
magv = dfile[:,2]
err_v = dfile[:,3]
magi = dfile[:,8]
err_i = dfile[:,9]

# color is V-I
color = magv - magi
# error summed in quadrature
err_color = (err_v**2 + err_i**2)**0.5

# make histogram
s = np.array(magv)
xmin = 10
xmax = 25
step = 0.1
y, x = np.histogram(s, bins=np.linspace(xmin, xmax, (xmax-xmin)/step))
nbins = y.size

def plot_histogram():
	plt.clf()
	plt.bar(x[:-1], y, width=x[1]-x[0], color='red', alpha=0.5)
	#plt.grid(True)
	plt.xlabel('mag_v')
	plt.ylabel('N')
	plt.savefig('histo_of_vmag.png')
# call function
plot_histogram()

def plot_VI_histo():
	global y_c, histx_c
	s_c = np.array(color)
	xmin_c = -5
	xmax_c = 5
	# steps chosen to have roughly same number of divisions as mag histograms
	step_c = 0.066
	y_c, x_c = np.histogram(s_c, bins=np.linspace(xmin_c, xmax_c, (xmax_c-xmin_c)/step_c))
	nbins_c = y.size
	plt.clf()
	plt.bar(x_c[:-1], y_c, width=x_c[1]-x_c[0], color='gray', alpha=0.5)
	#plt.grid(True)
	plt.xlabel('color')
	plt.ylabel('N')
	plt.savefig('histo_of_color.png')

	# make global variable for use later
	histx_c = x_c[:-1]
plot_VI_histo()

# makes x same dimensions as y
histx = x[:-1]

# Gaussian fit
def gaussian(x, a, b, c, d):
    '''
    a = (height of the peak - continuum), b = peak center, c = width of peak, d = continuum
    '''
    y = a * np.exp((-(x-b)**2)/(2*c**2)) + d
    return y

# fit gaussian to right peak
height_r = 1200
center_r = 20
width_r = 2
continuum_r = 0

# carry out fit
param_r, covar_r = optimized.curve_fit(gaussian, histx, y, p0=[height_r, center_r, width_r, continuum_r])
print param_r

# use parameters to generate fit
rightpeak = gaussian(histx, param_r[0], param_r[1], param_r[2], param_r[3])

def plot_hist_with_gauss():
	# plot histogram with fit
	plt.clf()
	plt.bar(x[:-1], y, width=x[1]-x[0], color='yellow', alpha=0.25)
	#plt.plot(histx, leftpeak, color='k', lw=2)
	plt.plot(histx, rightpeak, '.k', lw=2)
	plt.xlabel('mag_v')
	plt.ylabel('N')
	plt.savefig('mag_histo_with_fit.png')
plot_hist_with_gauss()

sig = param_r[2]

# determine mask params
mask1 = (histx < param_r[1]+sigma*sig) & (histx > param_r[1]-sigma*sig)
# use calculated center (above) not initially guessed value (line below)
#mask1 = (histx < center_r+0.5*sig) & (histx > center_r-0.5*sig)

# check by plotting
#plt.clf()
#plt.plot(histx[mask1], y[mask1], 'b.')

mask_sigv = (magv > min(histx[mask1])) & (magv < max(histx[mask1]))

# mask color and mag_v using params from gaussian fit to histogram
mask1_magv = magv[mask_sigv]
mask1_color = color[mask_sigv]
mask1_cerr = err_color[mask_sigv]
mask1_verr = err_v[mask_sigv]

#plt.clf()
#plt.errorbar(mask1_color, mask1_magv, xerr=mask1_cerr, yerr=mask1_verr, fmt='.')
#plt.ylim(22,16)


#------------------------------------------------------------------------------------------------
# fit gaussian to peak of color histogram
height_c = 4000
center_c = 0.5
width_c = 0.5
continuum_c = 0

paramc, covarc = optimized.curve_fit(gaussian, histx_c, y_c, p0=[height_c, center_c, width_c, continuum_c])
sigc = paramc[2]

maskc = (histx_c < paramc[1]+sigma*sigc) & (histx_c > paramc[1]-sigma*sigc)
mask_sigc = (color > min(histx_c[maskc])) & (color < max(histx_c[maskc]))

# apply mask from fitting gaussian to color histo
mask2_magv,mask2_color,mask2_cerr,mask2_verr = magv[mask_sigc],color[mask_sigc],err_color[mask_sigc],err_v[mask_sigc]

fmask_magv = np.append(mask1_magv, mask2_magv)
fmask_color = np.append(mask1_color, mask2_color)
fmask_cerr = np.append(mask1_cerr, mask2_cerr)
fmask_verr = np.append(mask1_verr, mask2_verr)

# plot to see how this restricted the cluster
plt.clf()
plt.errorbar(color, magv, xerr=err_color, yerr=err_v, fmt='.',alpha=0.2, color='red',label='original data')
plt.errorbar(fmask_color, fmask_magv, xerr=fmask_cerr, yerr=fmask_verr, fmt='.', alpha=0.2,color='blue',label='restricted data')
plt.ylim(22,13)
plt.xlabel('V-I')
plt.ylabel('mag_v')
plt.title('3$\sigma$ deviations of histogram gaussians')
plt.legend()
plt.savefig('field_overlay_'+str(sigma)+'sig.png')
